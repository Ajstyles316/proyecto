# Configuración de correo para envío automático de contraseñas
# Configura estas variables de entorno en tu sistema o servidor

# Configuración SMTP para Gmail
SMTP_SERVER=smtp.gmail.com
SMTP_PORT=587
SMTP_USERNAME=tu_email@gmail.com
SMTP_PASSWORD=tu_password_de_aplicacion

# Notas importantes:
# 1. Para Gmail, necesitas usar una "Contraseña de aplicación" en lugar de tu contraseña normal
# 2. Ve a: https://myaccount.google.com/apppasswords
# 3. Genera una contraseña de aplicación para "Correo"
# 4. Usa esa contraseña en SMTP_PASSWORD
# 5. Asegúrate de que la verificación en 2 pasos esté activada en tu cuenta de Google

# Para configurar en Windows (PowerShell):
# $env:SMTP_USERNAME="tu_email@gmail.com"
# $env:SMTP_PASSWORD="tu_password_de_aplicacion"

# Para configurar en Linux/Mac:
# export SMTP_USERNAME="tu_email@gmail.com"
# export SMTP_PASSWORD="tu_password_de_aplicacion"
